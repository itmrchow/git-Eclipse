--建立customer表格
create table CUSTOMER (
    C_ID varchar2(10) not null constraint customer_pk primary key,
    C_NAME varchar2(10) not null,
    SEX number not null,
    PHONENUM varchar2(4000),
    CREATEDATE date not null,
    MODIFYDATE date not null
);


--建立order表格
create table ORDERS (
    O_ID varchar2(10) not null constraint order_pk primary key,
    C_ID varchar2(10) not null,
    PRICE number,
    CREATEDATE date,
    MODIFYDATE date
);

--建立PRODUCTS表格
create table PRODUCTS (
    P_ID varchar2(10) not null constraint products_pk primary key,
    O_ID varchar2(10) not null,
    P_NAME varchar2(20) not null,
    CREATEDATE date not null,
    MODIFYDATE date not null
);
