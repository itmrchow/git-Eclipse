--DUAL
select sysdate from dual

--DATE 轉 CHAR
select 
    to_char(hiredate,'yyyy-mm-dd')
from 
    SCOTT.EMP

--CHAR轉DATE 
select 
    to_date('1993-01-20','yyyy-mm-dd')
from
    dual

--COUNT GROUP BY 
select 
    count(JOB) , JOB
from 
    SCOTT.EMP
GROUP BY JOB

--子查詢
select
    *
from
    (select * FROM SCOTT.EMP WHERE SAL >= 3000) res
WHERE
 RES.DEPTNO = '20'

--子查詢(SELECT)
SELECT
    EMP.* , 
    (SELECT DEPT.DNAME FROM SCOTT.DEPT DEPT WHERE EMP.DEPTNO = DEPT.DEPTNO ) AS DEPTNAME
FROM
    SCOTT.EMP EMP
WHERE
    DEPTNO in ('10','20')
    AND
    --EXISTS
    EXISTS (
        SELECT * FROM SCOTT.DEPT DEPT WHERE DEPT.DEPTNO = EMP.DEPTNO AND DEPT.LOC = 'DALLAS'
    )
    
     
--DECODE
select 
    EMPNO,
    ENAME,
    decode(JOB,'PRESIDENT','頭','MANAGER','經理','SALESMAN','業務','OTHER'),
    JOB AS JOB , 
    MGR
from SCOTT.EMP
